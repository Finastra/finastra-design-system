export const Template = ({ apps , appNameProperty, shortAppNameProperty, title, baseUrl, tenantId, channelType, tools }) => {
  return html`
    <fds-launchpad
      .apps=${apps}
      .appNameProperty=${appNameProperty}
      .shortAppNameProperty=${shortAppNameProperty}
      .title=${title}
      .baseUrl=${baseUrl}
      .channelType=${channelType}
    >
      ${tools ? html`<div slot='tools'>${tools}</div>` : ''}
    </fsd-launchpad>
  `;
};
