import { Meta, Story, Canvas, ArgsTable, Props } from '@storybook/addon-docs';
import { html } from 'lit';
import {cssprops, argTypes} from './custom-element.json';

import '@finastra/avatar';

<Meta
  title="Components/Avatar"
  component={'fds-avatar'}
  argTypes={argTypes}
  parameters={{
    cssprops
  }}
/>

export const Template = ({ name, shortName, dense = false, large = false, primary = false, secondary = false }) => {
  return html`<fds-avatar
    name=${name}
    shortName=${shortName}
    ?dense=${dense}
    ?large=${large}
    ?primary=${primary}
    ?secondary=${secondary}
  ></fds-avatar>`;
};
