import { Meta, Story, Canvas, ArgsTable, Props } from '@storybook/addon-docs';
import { html } from 'lit';
import {cssprops, argTypes} from './custom-element.json';

import '@finastra/app-card';

<Meta
  title="Components/App Card"
  component={'fds-app-card'}
  argTypes={argTypes}
  parameters={{
    cssprops
  }}
/>

export const Template = ({ secondary = false, large = false, extraDense = false, dense = false, label, shortLabel }) => {
  return html`<fds-app-card
    ?secondary=${secondary}
    ?large=${large}
    ?extraDense=${extraDense}
    ?dense=${dense}
    .label=${label}
    .shortLabel=${shortLabel}
  ></fds-app-card>`;
};
